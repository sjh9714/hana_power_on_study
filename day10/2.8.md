# 6. 가장 기본적인 이력 관리 코드

```python id="n1jlwm"
...
# --------------------------------------
# Chat Function
# --------------------------------------
def chat(message, history):
    # history 구조 출력합니다
    print(history)

    messages = []

    #이전 질문 및 답변을 추가합니다
    for item in history:
        messages.append({
            "role": item["role"],
            "content": item["content"][0]["text"]
        })

    # --------------------------------------
    # 현재 질문 추가
    # --------------------------------------
    messages.append({
        "role": "user",
        "content": message
    })

    # --------------------------------------
    # OpenAI 호출
    # --------------------------------------
    response = client.chat.completions.create(
        model="gpt-4.1-mini",
        messages=messages
    )

    answer = response.choices[0].message.content

    return answer
...

```
